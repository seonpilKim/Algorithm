vector<int> SieveOfEratosthenes() {
	vector<bool> numbers(N + 1, true);
	vector<int> primes;

	for (int i = 2; i <= N; i++) {
		if (!numbers[i])
			continue;

		for (int j = 2 * i; j <= N; j += i)
			numbers[j] = false;
	}

	for (int i = 2; i <= N; i++)
		if (numbers[i])
			primes.push_back(i);

	return primes;
}
