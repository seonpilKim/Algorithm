vector<vector<bool>> dp(N + 1, vector<bool>(N + 1));
// base condition
for (int i = 1; i <= N; i++) 
	dp[i][i] = true;
for (int i = 1; i < N; i++) 
	if (seq[i] == seq[i + 1])
		dp[i][i + 1] = true;

// bottom-up
for (int range = 3; range <= N; range++) {
	for (int start = 1; start + range - 1 <= N; start++) {
		int end = start + range - 1;

		if (seq[start] == seq[end] && dp[start + 1][end - 1] == true) 
			dp[start][end] = true;
	}
}
