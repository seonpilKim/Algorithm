// Top-down
void DFS(int idx, int sum) {
	if (idx > N || dp[idx][sum])
		return;

	dp[idx][sum] = true;

	DFS(idx + 1, sum + sinkers[idx]);
	DFS(idx + 1, sum);
	DFS(idx + 1, abs(sinkers[idx] - sum));
}
